Fungsi Datetime

MySQL juga menyediakan banyak sekali function yang bisa kita gunakan untuk mengolah data tipe Date dan Time

CURRENT_DATE()

Mengembalikan tanggal saat ini. 

        SELECT CURRENT_DATE();

Hasilnya 

        +----------------+
        | CURRENT_DATE() |
        +----------------+
        | 2024-07-11     |
        +----------------+

CURRENT_TIME()

Mengembalikan waktu saat ini. 

        SELECT CURRENT_TIME();

Hasilnya 

        +----------------+
        | CURRENT_TIME() |
        +----------------+
        | 14:30:00       |
        +----------------+

NOW()

Mengembalikan tanggal dan waktu saat ini. 

        SELECT NOW();

Hasilnya 

        +---------------------+
        | NOW()               |
        +---------------------+
        | 2024-07-11 14:30:00 |
        +---------------------+

DATE()

Mengembalikan bagian tanggal dari datetime atau timestamp. 

        SELECT DATE('2024-07-11 14:30:00');

Hasilnya 

        +-----------------------------+
        | DATE('2024-07-11 14:30:00') |
        +-----------------------------+
        | 2024-07-11                  |
        +-----------------------------+

TIME()

Mengembalikan bagian waktu dari datetime atau timestamp. 

        SELECT TIME('2024-07-11 14:30:00');

Hasilnya 

        +----------------------------+
        | TIME('2024-07-11 14:30:00') |
        +----------------------------+
        | 14:30:00                    |
    


    
YEAR()

    

        Mengembalikan tahun dari tanggal.
    

    
        SELECT YEAR('2024-07-11');
    

    

        Hasilnya
    

    
        +--------------------+
        | YEAR('2024-07-11') |
        +--------------------+
        | 2024               |
        +--------------------+
    


    
MONTH()

    
Mengembalikan bulan dari tanggal.

    
        SELECT MONTH('2024-07-11');
    

    

        Hasilnya
    

    
        +---------------------+
        | MONTH('2024-07-11') |
        +---------------------+
        | 7                   |
        +---------------------+
    

    


    
DAY()

    

        Mengembalikan hari dari tanggal.
    

    
        SELECT DAY('2024-07-11');
    

    

        Hasilnya
    

    
        +-------------------+
        | DAY('2024-07-11') |
        +-------------------+
        | 11                |
        +-------------------+
    


    
DATE_ADD()

    

        Menambahkan interval waktu ke tanggal.
    

    
        SELECT DATE_ADD('2024-07-11', INTERVAL 10 DAY);
    

    

        Hasilnya
    

    
        +-----------------------------------------+
        | DATE_ADD('2024-07-11', INTERVAL 10 DAY) |
        +-----------------------------------------+
        | 2024-07-21                              |
        +-----------------------------------------+
    


    
DATEDIFF()

    

        Menghitung selisih antara dua tanggal.
    

    
        SELECT DATEDIFF('2024-07-11', '2024-07-01');
    

    

        Hasilnya
    

    
        +--------------------------------------+
        | DATEDIFF('2024-07-11', '2024-07-01') |
        +--------------------------------------+
        | 10                                   |
    


    
STR_TO_DATE()

    

        Mengubah string menjadi tanggal sesuai format tertentu.
    

    
        SELECT STR_TO_DATE('11-07-2024', '%d-%m-%Y');
    

    

        Hasilnya
    

    
        +---------------------------------------+
        | STR_TO_DATE('11-07-2024', '%d-%m-%Y') |
        +---------------------------------------+
        | 2024-07-11                            |
        +---------------------------------------+
    


    
DATE_FORMAT()

    

        Mengubah tanggal menjadi string sesuai format tertentu.
    

    
        SELECT DATE_FORMAT('2024-07-11', '%d/%m/%Y');
    

    

        Hasilnya
    

    
        +---------------------------------+
        | DATE_FORMAT('2024-07-11', '%d/%m/%Y') |
        +---------------------------------+
        | 11/07/2024                      |
        +---------------------------------+
    


    
Fungsi Datetime Lain
   
    

        

          Nama
          Deskripsi
          Perintah
          Hasil
        

        

          ADDDATE()
          Menambahkan nilai waktu (interval) ke nilai tanggal
          SELECT ADDDATE('2024-07-11', INTERVAL 10 DAY);
          2024-07-21
        

        

          ADDTIME()
          Menambahkan waktu
          SELECT ADDTIME('14:30:00', '2:30:00');
          17:00:00
        

        

          CONVERT_TZ()
          Mengubah dari satu zona waktu ke zona waktu lain
          SELECT CONVERT_TZ('2024-07-11 14:30:00', 'UTC', 'Asia/Jakarta');
          2024-07-11 21:30:00
        

        

          CURDATE()
          Mengembalikan tanggal saat ini
          SELECT CURDATE();
          2024-07-11
        

        

          CURRENT_DATE(), CURRENT_DATE
          Sinonim untuk CURDATE()
          SELECT CURRENT_DATE();
          2024-07-11
        

        

          CURRENT_TIME(), CURRENT_TIME
          Sinonim untuk CURTIME()
          SELECT CURRENT_TIME();
          14:30:00
        

        

          CURRENT_TIMESTAMP(), CURRENT_TIMESTAMP
          Sinonim untuk NOW()
          SELECT CURRENT_TIMESTAMP();
          2024-07-11 14:30:00
        

        

          CURTIME()
          Mengembalikan waktu saat ini
          SELECT CURTIME();
          14:30:00
        

        

          DATE()
          Mengekstrak bagian tanggal dari ekspresi tanggal atau datetime
          SELECT DATE('2024-07-11 14:30:00');
          2024-07-11
        

        

          DATE_ADD()
          Menambahkan nilai waktu (interval) ke nilai tanggal
          SELECT DATE_ADD('2024-07-11', INTERVAL 10 DAY);
          2024-07-21
        

        

          DATE_FORMAT()
          Memformat tanggal sesuai spesifikasi
          SELECT DATE_FORMAT('2024-07-11', '%d/%m/%Y');
          11/07/2024
        

        

          DATE_SUB()
          Mengurangi nilai waktu (interval) dari tanggal
          SELECT DATE_SUB('2024-07-11', INTERVAL 10 DAY);
          2024-07-01
        

        

          DATEDIFF()
          Mengurangi dua tanggal
          SELECT DATEDIFF('2024-07-11', '2024-07-01');
          10
        

        

          DAY()
          Sinonim untuk DAYOFMONTH()
          SELECT DAY('2024-07-11');
          11
        

        

          DAYNAME()
          Mengembalikan nama hari dalam minggu
          SELECT DAYNAME('2024-07-11');
          Kamis
        

        

          DAYOFMONTH()
          Mengembalikan hari dalam bulan (0-31)
          SELECT DAYOFMONTH('2024-07-11');
          11
        

        

          DAYOFWEEK()
          Mengembalikan indeks hari dalam minggu dari argumen
          SELECT DAYOFWEEK('2024-07-11');
          5
        

        

          DAYOFYEAR()
          Mengembalikan hari dalam tahun (1-366)
          SELECT DAYOFYEAR('2024-07-11');
          193
        

        

          EXTRACT()
          Mengekstrak bagian dari tanggal
          SELECT EXTRACT(YEAR FROM '2024-07-11');
          2024
        

        

          FROM_DAYS()
          Mengubah nomor hari menjadi tanggal
          SELECT FROM_DAYS(738000);
          2023-11-14
        

        

          FROM_UNIXTIME()
          Memformat Unix timestamp sebagai tanggal
          SELECT FROM_UNIXTIME(1626000000);
          2021-07-11 12:00:00
        

        

          GET_FORMAT()
          Mengembalikan string format tanggal
          SELECT GET_FORMAT(DATE, 'EUR');
          %d.%m.%Y
        

        

          HOUR()
          Mengekstrak jam
          SELECT HOUR('14:30:00');
          14
        

        

          LAST_DAY
          Mengembalikan hari terakhir dalam bulan untuk argumen
          SELECT LAST_DAY('2024-07-11');
          2024-07-31
        

        

          LOCALTIME(), LOCALTIME
          Sinonim untuk NOW()
          SELECT LOCALTIME();
          2024-07-11 14:30:00
        

        

          LOCALTIMESTAMP, LOCALTIMESTAMP()
          Sinonim untuk NOW()
          SELECT LOCALTIMESTAMP();
          2024-07-11 14:30:00
        

        

          MAKEDATE()
          Membuat tanggal dari tahun dan hari dalam tahun
          SELECT MAKEDATE(2024, 193);
          2024-07-11
        

        

          MAKETIME()
          Membuat waktu dari jam, menit, detik
          SELECT MAKETIME(14, 30, 0);
          14:30:00
        

        

          MICROSECOND()
          Mengembalikan mikrodetik dari argumen
          SELECT MICROSECOND('14:30:00.123456');
          123456
        

        

          MINUTE()
          Mengekstrak menit dari argumen
          SELECT MINUTE('14:30:00');
          30
        

        

          MONTH()
          Mengembalikan bulan dari tanggal yang diberikan
          SELECT MONTH('2024-07-11');
          7
        

        

          MONTHNAME()
          Mengembalikan nama bulan dari tanggal yang diberikan
          SELECT MONTHNAME('2024-07-11');
          Juli
        

        

          NOW()
          Mengembalikan tanggal dan waktu saat ini
          SELECT NOW();
          2024-07-11 14:30:00
        

        

          PERIOD_ADD()
          Menambahkan periode ke tahun-bulan
          SELECT PERIOD_ADD(202407, 3);
          202410
        

        

          PERIOD_DIFF()
          Mengembalikan jumlah bulan antara dua periode
          SELECT PERIOD_DIFF(202407, 202404);
          3
        

        

          QUARTER()
          Mengembalikan kuartal dari tanggal yang diberikan
          SELECT QUARTER('2024-07-11');
          3
        

        

          SEC_TO_TIME()
          Mengkonversi detik menjadi format 'hh:mm:ss'
          SELECT SEC_TO_TIME(3661);
          01:01:01
        

        

          SECOND()
          Mengekstrak detik dari argumen
          SELECT SECOND('14:30:45');
          45
        

        

          STR_TO_DATE()
          Mengkonversi string menjadi tanggal
          SELECT STR_TO_DATE('10-12-2024', '%d-%m-%Y');
          2024-12-10
        

        

          SUBDATE()
          Sinonim untuk DATE_SUB() ketika dipanggil dengan tiga argumen
          SELECT SUBDATE('2024-07-11', INTERVAL 10 DAY);
          2024-07-01
        

        

          SUBTIME()
          Mengurangi waktu
          SELECT SUBTIME('14:30:00', '2:30:00');
          12:00:00
        

        

          SYSDATE()
          Mengembalikan waktu eksekusi fungsi
          SELECT SYSDATE();
          2024-07-11 14:30:00
        

        

          TIME()
          Mengekstrak bagian waktu dari ekspresi yang diberikan
          SELECT TIME('2024-07-11 14:30:00');
          14:30:00
        

        

          TIME_FORMAT()
          Memformat waktu sesuai spesifikasi
          SELECT TIME_FORMAT('14:30:00', '%H:%i:%s');
          14:30:00
        

        

          TIME_TO_SEC()
          Mengkonversi waktu menjadi detik
          SELECT TIME_TO_SEC('14:30:00');
          52200
        

        

          TIMEDIFF()
          Mengurangi waktu
          SELECT TIMEDIFF('14:30:00', '12:45:00');
          01:45:00
        

        

          TIMESTAMP()
          Dengan satu argumen, fungsi ini mengembalikan ekspresi tanggal atau datetime; dengan dua argumen, jumlah dari argumen tersebut
          SELECT TIMESTAMP('2024-07-11');
          2024-07-11 00:00:00
        

        

          TIMESTAMPADD()
          Menambahkan interval ke ekspresi datetime
          SELECT TIMESTAMPADD(MINUTE, 30, '2024-07-11 14:30:00');
          2024-07-11 15:00:00
        

        

          TIMESTAMPDIFF()
          Mengembalikan selisih dari dua ekspresi datetime dengan satuan yang ditentukan
          SELECT TIMESTAMPDIFF(MINUTE, '2024-07-11 14:00:00', '2024-07-11 15:30:00');
          90
        

        

          TO_DAYS()
          Mengkonversi argumen tanggal menjadi hari
          SELECT TO_DAYS('2024-07-11');
          738000
        

        

          TO_SECONDS()
          Mengkonversi argumen tanggal atau datetime menjadi detik sejak Tahun 0
          SELECT TO_SECONDS('2024-07-11 14:30:00');
          67879950000
        

        

          UNIX_TIMESTAMP()
          Mengembalikan timestamp Unix
          SELECT UNIX_TIMESTAMP('2024-07-11 14:30:00');
          1718746200
        

        

          UTC_DATE()
          Mengembalikan tanggal UTC saat ini
          SELECT UTC_DATE();
          2024-07-11
        

        

          UTC_TIME()
          Mengembalikan waktu UTC saat ini
          SELECT UTC_TIME();
          07:30:00
        

        

          UTC_TIMESTAMP()
          Mengembalikan tanggal dan waktu UTC saat ini
          SELECT UTC_TIMESTAMP();
          2024-07-11 07:30:00
        

        

          WEEK()
          Mengembalikan nomor minggu dalam tahun
          SELECT WEEK('2024-07-11');
          28
        

        

          WEEKDAY()
          Mengembalikan indeks hari dalam minggu (0 untuk Senin, 6 untuk Minggu)
          SELECT WEEKDAY('2024-07-11');
          4
        

        

          WEEKOFYEAR()
          Mengembalikan minggu kalender dalam tahun
          SELECT WEEKOFYEAR('2024-07-11');
          28
        

        

          YEAR()
          Mengembalikan tahun dari tanggal yang diberikan
          SELECT YEAR('2024-07-11');
          2024
        

        

          YEARWEEK()
          Mengembalikan tahun dan minggu dalam format YYYYWW
          SELECT YEARWEEK('2024-07-11');
          202428
